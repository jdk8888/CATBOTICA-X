Extract Design from catbotica.com

Instructions:

  1. Open catbotica.com in a new tab
  2. Open Developer Tools (F12)
  3. Go to the Console tab
  4. Copy and paste the code below into the console
  5. Press Enter
  6. Copy the results and share them

Code to paste in Console:

// Extract fonts, colors, and design from catbotica.com const styles = window.getComputedStyle(document.body); const fonts = []; const colors = new Set(); const images = []; // Get all fonts document.querySelectorAll('*').forEach(el => { const fontFamily = window.getComputedStyle(el).fontFamily; if (fontFamily && fontFamily !== 'initial') { fonts.push(fontFamily.split(',')[0].replace(/['"]/g, '')); } }); // Get all colors document.querySelectorAll('*').forEach(el => { const bg = window.getComputedStyle(el).backgroundColor; const color = window.getComputedStyle(el).color; const border = window.getComputedStyle(el).borderColor; if (bg && bg !== 'rgba(0, 0, 0, 0)' && bg !== 'transparent') colors.add(bg); if (color && color !== 'rgba(0, 0, 0, 0)') colors.add(color); if (border && border !== 'rgba(0, 0, 0, 0)' && border !== 'transparent') colors.add(border); }); // Get all images document.querySelectorAll('img').forEach(img => { if (img.src) images.push(img.src); }); // Get logo const logo = document.querySelector('img[alt*="logo" i], img[src*="logo" i], .logo img, header img'); const logoUrl = logo ? logo.src : 'Not found'; // Get CSS variables const rootStyles = getComputedStyle(document.documentElement); const cssVars = {}; for (let i = 0; i < rootStyles.length; i++) { const prop = rootStyles[i]; if (prop.startsWith('--')) { cssVars[prop] = rootStyles.getPropertyValue(prop); } } const result = { fonts: [...new Set(fonts)].slice(0, 5), colors: Array.from(colors).slice(0, 20), logo: logoUrl, images: images.slice(0, 10), cssVariables: cssVars, bodyBackground: styles.backgroundColor, bodyColor: styles.color, bodyFont: styles.fontFamily }; console.log('=== CATBOTICA DESIGN EXTRACTION ==='); console.log(JSON.stringify(result, null, 2)); console.log('=== COPY THE ABOVE JSON ==='); // Also display in alert for easy copying const resultText = JSON.stringify(result, null, 2); const textarea = document.createElement('textarea'); textarea.value = resultText; textarea.style.position = 'fixed'; textarea.style.opacity = '0'; document.body.appendChild(textarea); textarea.select(); document.execCommand('copy'); document.body.removeChild(textarea); alert('Design data copied to clipboard! Also check console for full details.');
